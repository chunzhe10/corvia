# M2.1 Workspace + Devcontainer Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Add multi-repo workspace support to Corvia's CLI and config, then create a devcontainer-based demo repo (`corvia-workspace`) that showcases Corvia indexing its own codebase across two repos.

**Architecture:** Two parallel streams converging on a shared config format. Stream 1 extends `corvia-common` config structs and `corvia-cli` with a `workspace` subcommand group. Stream 2 creates the `corvia-workspace` repo with devcontainer, `corvia.toml`, and pre-ingested knowledge. The config format (`[workspace]` section in `corvia.toml`) is the keystone — designed and tested first.

**Tech Stack:** Rust (clap 4 derive, toml 0.8, serde, tokio, anyhow), git2 (repo cloning), existing GitAdapter + LiteStore + OllamaEngine. Devcontainer spec (JSON), Dockerfile, shell scripts.

**Design doc:** `docs/rfcs/2026-03-01-m2.1-workspace-devcontainer-design.md`

```rust
use anyhow::Result;
use corvia_common::config::{CorviaConfig, WorkspaceConfig, RepoConfig};
use std::path::{Path, PathBuf};

/// Derive a repo name from a URL (e.g., "https://github.com/org/my-repo" -> "my-repo")
pub fn repo_name_from_url(url: &str) -> String {
    url.trim_end_matches('/')
        .rsplit('/')
        .next()
        .unwrap_or("unknown")
        .trim_end_matches(".git")
        .to_string()
}

/// Parse local overrides from "name=path" format
pub fn parse_local_overrides(locals: &[String]) -> Result<Vec<(String, String)>> {
    locals
        .iter()
        .map(|s| {
            let parts: Vec<&str> = s.splitn(2, '=').collect();
            if parts.len() != 2 {
                anyhow::bail!("Invalid local override format: '{}'. Expected 'name=/path/to/repo'", s);
            }
            Ok((parts[0].to_string(), parts[1].to_string()))
        })
        .collect()
}

/// Generate a workspace config from CLI arguments
pub fn generate_workspace_config(
    name: &str,
    repo_urls: &[String],
    local_overrides: &[(String, String)],
) -> CorviaConfig {
    let repos: Vec<RepoConfig> = repo_urls
        .iter()
        .map(|url| {
            let repo_name = repo_name_from_url(url);
            let local = local_overrides
                .iter()
                .find(|(n, _)| n == &repo_name)
                .map(|(_, p)| p.clone());
            RepoConfig {
                name: repo_name.clone(),
                url: url.clone(),
                local,
                namespace: repo_name,
            }
        })
        .collect();

    let mut config = CorviaConfig::default();
    config.project.name = name.to_string();
    config.project.scope_id = name.to_string();
    config.workspace = Some(WorkspaceConfig {
        repos_dir: "repos".into(),
        repos,
    });
    config
}

/// Resolve the actual path for a repo: local override if valid, otherwise cloned path
pub fn resolve_repo_path(workspace_root: &Path, repos_dir: &str, repo: &RepoConfig) -> PathBuf {
    if let Some(local) = &repo.local {
        let local_path = PathBuf::from(local);
        if local_path.exists() && local_path.join(".git").exists() {
            return local_path;
        }
    }
    workspace_root.join(repos_dir).join(&repo.name)
}

/// Clone a repo into the workspace repos directory. Returns the clone path.
pub fn clone_repo(url: &str, target_dir: &Path) -> Result<PathBuf> {
    if target_dir.exists() {
        println!("  repo already exists: {}", target_dir.display());
        return Ok(target_dir.to_path_buf());
    }
    println!("  cloning {} -> {}", url, target_dir.display());
    git2::Repository::clone(url, target_dir)?;
    Ok(target_dir.to_path_buf())
}

/// Scaffold the workspace directory structure
pub fn scaffold_workspace(root: &Path, config: &CorviaConfig) -> Result<()> {
    let ws = config.workspace.as_ref()
        .ok_or_else(|| anyhow::anyhow!("Not a workspace config"))?;

    // Create workspace root
    std::fs::create_dir_all(root)?;

    // Create repos directory
    let repos_dir = root.join(&ws.repos_dir);
    std::fs::create_dir_all(&repos_dir)?;

    // Create .gitkeep in repos dir
    std::fs::write(repos_dir.join(".gitkeep"), "")?;

    // Create .corvia directory
    let corvia_dir = root.join(&config.storage.data_dir);
    std::fs::create_dir_all(&corvia_dir)?;

    // Write corvia.toml
    let config_path = root.join("corvia.toml");
    config.save(&config_path)?;

    // Write .gitignore
    let gitignore = root.join(".gitignore");
    std::fs::write(
        &gitignore,
        "# Cloned repos (managed by corvia workspace init)\nrepos/\n\n# Ephemeral indexes (rebuilt from knowledge files)\n.corvia/hnsw_index/\n.corvia/coordination.redb\n.corvia/coordination.redb.lock\n",
    )?;

    Ok(())
}
```

```rust
/// Initialize a workspace: clone missing repos, set up .corvia/, provision Ollama
pub async fn init_workspace(root: &Path) -> Result<()> {
    let config_path = root.join("corvia.toml");
    if !config_path.exists() {
        anyhow::bail!("No corvia.toml found in {}. Run 'corvia workspace create' first.", root.display());
    }
    let config = CorviaConfig::load(&config_path)?;
    let ws = config.workspace.as_ref()
        .ok_or_else(|| anyhow::anyhow!("corvia.toml has no [workspace] section"))?;

    // Clone or link repos
    println!("Initializing workspace repos...");
    for repo in &ws.repos {
        let repo_path = resolve_repo_path(root, &ws.repos_dir, repo);
        if repo_path.exists() {
            println!("  {} — already present at {}", repo.name, repo_path.display());
        } else {
            clone_repo(&repo.url, &repo_path)?;
        }
    }

    // Initialize .corvia store
    println!("Initializing knowledge store...");
    let data_dir = root.join(&config.storage.data_dir);
    std::fs::create_dir_all(&data_dir)?;

    let store = corvia_kernel::create_store_at(&config, &data_dir).await?;
    store.init_schema().await?;

    // Write .corvia/.gitignore for ephemeral files
    let corvia_gitignore = data_dir.join(".gitignore");
    if !corvia_gitignore.exists() {
        std::fs::write(
            &corvia_gitignore,
            "hnsw_index/\ncoordination.redb\ncoordination.redb.lock\nstaging/\n",
        )?;
    }

    // Provision Ollama if using it
    if config.embedding.provider == corvia_common::config::InferenceProvider::Ollama {
        println!("Checking Ollama...");
        let provisioner = corvia_kernel::ollama_provisioner::OllamaProvisioner::new(
            &config.embedding.url,
        );
        provisioner.ensure_ready(&config.embedding.model).await?;
    }

    println!("Workspace initialized.");
    Ok(())
}
```

```rust
/// Create a store at a specific directory path (for workspace support)
pub async fn create_store_at(
    config: &CorviaConfig,
    data_dir: &std::path::Path,
) -> Result<Box<dyn QueryableStore>> {
    match config.storage.store_type {
        StoreType::Lite => {
            let store = lite_store::LiteStore::open(data_dir, config.embedding.dimensions)?;
            Ok(Box::new(store))
        }
        StoreType::Surrealdb => {
            // FullStore uses URL, not local path — pass through to existing factory
            create_store(config).await
        }
    }
}
```

```rust
/// Add a repo to the workspace config
pub fn add_repo_to_config(
    config: &mut CorviaConfig,
    url: &str,
    name: Option<&str>,
    namespace: Option<&str>,
    local: Option<&str>,
) -> Result<()> {
    let ws = config.workspace.as_mut()
        .ok_or_else(|| anyhow::anyhow!("Not a workspace config"))?;

    let repo_name = name.map(String::from).unwrap_or_else(|| repo_name_from_url(url));
    if ws.repos.iter().any(|r| r.name == repo_name) {
        anyhow::bail!("Repo '{}' already exists in workspace", repo_name);
    }

    ws.repos.push(RepoConfig {
        name: repo_name.clone(),
        url: url.to_string(),
        local: local.map(String::from),
        namespace: namespace.map(String::from).unwrap_or(repo_name),
    });
    Ok(())
}

/// Remove a repo from the workspace config
pub fn remove_repo_from_config(config: &mut CorviaConfig, name: &str) -> Result<()> {
    let ws = config.workspace.as_mut()
        .ok_or_else(|| anyhow::anyhow!("Not a workspace config"))?;

    let before = ws.repos.len();
    ws.repos.retain(|r| r.name != name);
    if ws.repos.len() == before {
        anyhow::bail!("Repo '{}' not found in workspace", name);
    }
    Ok(())
}
```

```rust
use corvia_adapter_git::GitAdapter;
use corvia_kernel::traits::{QueryableStore, InferenceEngine};
use corvia_common::config::InferenceProvider;

const EMBED_BATCH_SIZE: usize = 32;

/// Ingest all (or one) workspace repos with namespace isolation
pub async fn ingest_workspace(
    root: &Path,
    repo_filter: Option<&str>,
    fresh: bool,
) -> Result<()> {
    let config_path = root.join("corvia.toml");
    let config = CorviaConfig::load(&config_path)?;
    let ws = config.workspace.as_ref()
        .ok_or_else(|| anyhow::anyhow!("Not a workspace — no [workspace] section in corvia.toml"))?;

    let data_dir = root.join(&config.storage.data_dir);
    let store = corvia_kernel::create_store_at(&config, &data_dir).await?;
    let engine = corvia_kernel::create_engine(&config);

    let repos_to_ingest: Vec<&RepoConfig> = if let Some(name) = repo_filter {
        let repo = ws.repos.iter().find(|r| r.name == name)
            .ok_or_else(|| anyhow::anyhow!("Repo '{}' not found in workspace config", name))?;
        vec![repo]
    } else {
        ws.repos.iter().collect()
    };

    for repo_config in &repos_to_ingest {
        let repo_path = resolve_repo_path(root, &ws.repos_dir, repo_config);
        if !repo_path.exists() {
            println!("  Skipping {} — not cloned (run corvia workspace init)", repo_config.name);
            continue;
        }

        println!("\nIngesting {} (namespace: {})...", repo_config.name, repo_config.namespace);

        if fresh {
            println!("  Clearing existing entries for scope {}...", config.project.scope_id);
            // Note: fresh mode for a specific namespace would need scope+workstream delete
            // For now, this is per-repo re-ingest (entries are replaced by new versions)
        }

        let adapter = GitAdapter::new();
        let repo_path_str = repo_path.to_str()
            .ok_or_else(|| anyhow::anyhow!("Invalid path for repo {}", repo_config.name))?;
        let mut entries = adapter.ingest(repo_path_str).await?;

        println!("  {} entries from tree-sitter", entries.len());

        // Set scope and workstream for namespace isolation
        for entry in &mut entries {
            entry.scope_id = config.project.scope_id.clone();
            entry.workstream = repo_config.namespace.clone();
        }

        // Batch embed
        let texts: Vec<String> = entries.iter().map(|e| e.content.clone()).collect();
        let mut embedded_count = 0;
        for chunk in texts.chunks(EMBED_BATCH_SIZE) {
            let chunk_vec: Vec<String> = chunk.to_vec();
            let embeddings = engine.embed_batch(&chunk_vec).await?;
            for (i, embedding) in embeddings.into_iter().enumerate() {
                entries[embedded_count + i].embedding = Some(embedding);
            }
            embedded_count += chunk.len();
            println!("  embedded {}/{}", embedded_count, entries.len());
        }

        // Insert
        for entry in &entries {
            store.insert(entry).await?;
        }
        println!("  {} entries stored", entries.len());
    }

    println!("\nWorkspace ingest complete.");
    Ok(())
}
```

```rust
/// Display workspace status
pub async fn workspace_status(root: &Path) -> Result<()> {
    let config_path = root.join("corvia.toml");
    let config = CorviaConfig::load(&config_path)?;
    let ws = config.workspace.as_ref()
        .ok_or_else(|| anyhow::anyhow!("Not a workspace"))?;

    println!("Workspace: {} (scope: {})", config.project.name, config.project.scope_id);
    println!("Store: {:?} at {}", config.storage.store_type, config.storage.data_dir);
    println!("Embedding: {:?} ({})", config.embedding.provider, config.embedding.model);
    println!();

    // Try to open store for entry counts
    let data_dir = root.join(&config.storage.data_dir);
    let store = if data_dir.exists() {
        corvia_kernel::create_store_at(&config, &data_dir).await.ok()
    } else {
        None
    };

    println!("Repos ({}):", ws.repos.len());
    for repo in &ws.repos {
        let repo_path = resolve_repo_path(root, &ws.repos_dir, repo);
        let exists = repo_path.exists();
        let is_local = repo.local.as_ref()
            .map(|l| PathBuf::from(l).exists())
            .unwrap_or(false);

        let source = if is_local { "local" } else if exists { "cloned" } else { "missing" };

        // Get entry count for this namespace if store is available
        let count = if let Some(ref store) = store {
            // count returns total for scope_id; we'd ideally filter by workstream
            // For now, show total scope count (workspace-level)
            // TODO: add count_by_workstream to QueryableStore trait
            None::<u64>
        } else {
            None
        };

        let count_str = count
            .map(|c| format!(" ({} entries)", c))
            .unwrap_or_default();

        println!("  {} [{}] namespace:{}{}", repo.name, source, repo.namespace, count_str);
        println!("    url: {}", repo.url);
        if let Some(local) = &repo.local {
            println!("    local: {}", local);
        }
        println!("    path: {}", repo_path.display());
    }

    if let Some(ref store) = store {
        let total = store.count(&config.project.scope_id).await.unwrap_or(0);
        println!("\nTotal knowledge entries: {}", total);
    }

    Ok(())
}
```

**Step 3: Write post-create.sh**

```bash
#!/bin/bash
set -e

echo "=== Corvia Workspace: Post-Create Setup ==="

# Clone workspace repos
echo "Cloning repos..."
corvia workspace init || {
    echo "corvia not installed yet — building from source..."
    cd repos/corvia 2>/dev/null || {
        git clone https://github.com/anthropics/corvia repos/corvia
    }
    cd /workspace/repos/corvia
    cargo install --path crates/corvia-cli
    cd /workspace
    corvia workspace init
}

# Pull Ollama model
echo "Pulling embedding model..."
ollama serve &
sleep 3
ollama pull nomic-embed-text
pkill ollama || true

echo "=== Post-Create Complete ==="
```

**Step 4: Write post-start.sh**

```bash
#!/bin/bash
set -e

echo "=== Corvia Workspace: Starting Services ==="

# Start Ollama in background
ollama serve &
sleep 2

# Start Corvia server
corvia serve --mcp &

echo "Corvia server running on http://localhost:8020"
echo "Try: corvia search 'how does embedding work'"
```

**Step 5: Make scripts executable and commit**

```bash
chmod +x /root/corvia-workspace/.devcontainer/scripts/*.sh
cd /root/corvia-workspace

```
