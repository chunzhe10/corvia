# M3.2 RAG Pipeline Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Add a trait-based RAG pipeline (Retriever → Augmenter → GenerationEngine) to corvia-kernel with REST/MCP endpoints.

**Architecture:** Three pluggable trait stages orchestrated by `RagPipeline`. Ships with `VectorRetriever`, `GraphExpandRetriever`, `StructuredAugmenter`. Two modes: `context()` (R+A) and `ask()` (R+A+G). All search routed through pipeline. D23/D45-aligned observability.

**Tech Stack:** Rust, async-trait, corvia-kernel (hnsw_rs, petgraph, Redb), corvia-server (axum), tracing, uuid, chrono, serde

**Design doc:** `docs/rfcs/2026-03-02-m3.2-rag-pipeline-design.md`

```rust
/// Pure vector search retriever. Baseline strategy.
/// Embeds query → store.search() → RBAC filter → visibility filter.
pub struct VectorRetriever {
    store: Arc<dyn QueryableStore>,
    engine: Arc<dyn InferenceEngine>,
}

impl VectorRetriever {
    pub fn new(store: Arc<dyn QueryableStore>, engine: Arc<dyn InferenceEngine>) -> Self {
        Self { store, engine }
    }
}

#[async_trait]
impl Retriever for VectorRetriever {
    fn name(&self) -> &str { "vector" }

    async fn retrieve(
        &self,
        query: &str,
        scope_id: &str,
        opts: &RetrievalOpts,
    ) -> Result<RetrievalResult> {
        let start = Instant::now();

        // RBAC check
        if let Some(ref perms) = opts.permissions {
            if let AgentPermission::ReadWrite { scopes } = perms {
                if !scopes.iter().any(|s| s == scope_id || s == "*") {
                    return Ok(RetrievalResult {
                        results: Vec::new(),
                        metrics: RetrievalMetrics {
                            latency_ms: start.elapsed().as_millis() as u64,
                            vector_results: 0,
                            graph_expanded: 0,
                            post_filter_count: 0,
                            retriever_name: self.name().into(),
                        },
                    });
                }
            }
        }

        let _span = tracing::info_span!("corvia.rag.retrieve.vector",
            retriever = self.name(),
            scope_id = scope_id,
        ).entered();

        // Embed query
        let embedding = self.engine.embed(query).await?;

        // 2x oversample for post-filtering
        let fetch_limit = std::cmp::max(opts.limit * 2, 10);
        let results = self.store.search(&embedding, scope_id, fetch_limit).await?;
        let vector_results = results.len();

        // Visibility post-filter
        let filtered: Vec<SearchResult> = results
            .into_iter()
            .filter(|r| visibility_filter(r, &opts.visibility, opts.agent_id.as_deref()))
            .take(opts.limit)
            .collect();

        let post_filter_count = filtered.len();

        info!(
            vector_results,
            post_filter_count,
            latency_ms = start.elapsed().as_millis() as u64,
            "rag_retrieved"
        );

        Ok(RetrievalResult {
            results: filtered,
            metrics: RetrievalMetrics {
                latency_ms: start.elapsed().as_millis() as u64,
                vector_results,
                graph_expanded: 0,
                post_filter_count,
                retriever_name: self.name().into(),
            },
        })
    }
}

/// Shared visibility filter logic (extracted from ContextBuilder for reuse).
fn visibility_filter(r: &SearchResult, visibility: &VisibilityMode, agent_id: Option<&str>) -> bool {
    match r.entry.entry_status {
        EntryStatus::Merged => true,
        EntryStatus::Pending | EntryStatus::Committed => {
            match visibility {
                VisibilityMode::Own => {
                    match (agent_id, &r.entry.agent_id) {
                        (Some(aid), Some(entry_aid)) => aid == entry_aid,
                        _ => false,
                    }
                }
                VisibilityMode::All => true,
                VisibilityMode::Explicit(agents) => {
                    match &r.entry.agent_id {
                        Some(entry_aid) => agents.iter().any(|a| a == entry_aid),
                        None => false,
                    }
                }
            }
        }
        EntryStatus::Rejected => false,
    }
}
```

```rust
/// Vector + graph expansion retriever. Corvia's differentiator.
/// 1. Vector search for initial top-k
/// 2. For each result, follow graph edges up to graph_depth hops
/// 3. Deduplicate and blend scores: final = (1-α)*cosine + α*(1/(hop+1))
pub struct GraphExpandRetriever {
    store: Arc<dyn QueryableStore>,
    engine: Arc<dyn InferenceEngine>,
    graph: Arc<dyn GraphStore>,
    alpha: f32,
}

impl GraphExpandRetriever {
    pub fn new(
        store: Arc<dyn QueryableStore>,
        engine: Arc<dyn InferenceEngine>,
        graph: Arc<dyn GraphStore>,
        alpha: f32,
    ) -> Self {
        Self { store, engine, graph, alpha }
    }
}

#[async_trait]
impl Retriever for GraphExpandRetriever {
    fn name(&self) -> &str { "graph_expand" }

    async fn retrieve(
        &self,
        query: &str,
        scope_id: &str,
        opts: &RetrievalOpts,
    ) -> Result<RetrievalResult> {
        let start = Instant::now();

        // RBAC check
        if let Some(ref perms) = opts.permissions {
            if let AgentPermission::ReadWrite { scopes } = perms {
                if !scopes.iter().any(|s| s == scope_id || s == "*") {
                    return Ok(RetrievalResult {
                        results: Vec::new(),
                        metrics: RetrievalMetrics {
                            latency_ms: start.elapsed().as_millis() as u64,
                            vector_results: 0,
                            graph_expanded: 0,
                            post_filter_count: 0,
                            retriever_name: self.name().into(),
                        },
                    });
                }
            }
        }

        let _span = tracing::info_span!("corvia.rag.retrieve",
            retriever = self.name(),
            scope_id = scope_id,
        ).entered();

        // Step 1: Vector search
        let embedding = self.engine.embed(query).await?;
        let fetch_limit = std::cmp::max(opts.limit * 2, 10);
        let vector_results = self.store.search(&embedding, scope_id, fetch_limit).await?;
        let vector_count = vector_results.len();

        // Track seen IDs and their best scores
        let mut seen: HashSet<uuid::Uuid> = HashSet::new();
        let mut scored: Vec<(SearchResult, f32)> = Vec::new();

        for r in &vector_results {
            seen.insert(r.entry.id);
            // Pure vector score (graph component = 0 for direct hits at hop 0)
            let final_score = (1.0 - self.alpha) * r.score + self.alpha * 1.0;
            scored.push((r.clone(), final_score));
        }

        // Step 2: Graph expansion (if enabled)
        let mut graph_expanded = 0;
        if opts.expand_graph {
            let _gs = tracing::info_span!("corvia.rag.retrieve.graph").entered();
            for r in &vector_results {
                let edges = self.graph.edges(&r.entry.id, EdgeDirection::Both).await?;
                for edge in edges {
                    let neighbor_id = if edge.from == r.entry.id { edge.to } else { edge.from };
                    if seen.contains(&neighbor_id) {
                        continue;
                    }
                    seen.insert(neighbor_id);

                    if let Ok(Some(entry)) = self.store.get(&neighbor_id).await {
                        if entry.scope_id != scope_id {
                            continue;
                        }
                        let hop_score = self.alpha * (1.0 / 2.0); // hop distance = 1
                        let result = SearchResult { entry, score: hop_score };
                        scored.push((result, hop_score));
                        graph_expanded += 1;
                    }
                }

                // Depth > 1: follow edges of edges
                if opts.graph_depth > 1 {
                    let traversed = self.graph.traverse(
                        &r.entry.id,
                        None,
                        EdgeDirection::Both,
                        opts.graph_depth,
                    ).await?;
                    for entry in traversed {
                        if seen.contains(&entry.id) || entry.scope_id != scope_id {
                            continue;
                        }
                        seen.insert(entry.id);
                        let hop_score = self.alpha * (1.0 / (opts.graph_depth as f32 + 1.0));
                        let result = SearchResult { entry, score: hop_score };
                        scored.push((result, hop_score));
                        graph_expanded += 1;
                    }
                }
            }
        }

        // Step 3: Sort by blended score, apply visibility filter, take limit
        scored.sort_by(|a, b| b.1.partial_cmp(&a.1).unwrap_or(std::cmp::Ordering::Equal));

        let _fs = tracing::info_span!("corvia.rag.retrieve.filter").entered();
        let filtered: Vec<SearchResult> = scored
            .into_iter()
            .map(|(r, _score)| r)
            .filter(|r| visibility_filter(r, &opts.visibility, opts.agent_id.as_deref()))
            .take(opts.limit)
            .collect();

        let post_filter_count = filtered.len();

        info!(
            vector_count,
            graph_expanded,
            post_filter_count,
            latency_ms = start.elapsed().as_millis() as u64,
            "rag_retrieved"
        );

        Ok(RetrievalResult {
            results: filtered,
            metrics: RetrievalMetrics {
                latency_ms: start.elapsed().as_millis() as u64,
                vector_results: vector_count,
                graph_expanded,
                post_filter_count,
                retriever_name: self.name().into(),
            },
        })
    }
}
```

```rust
const DEFAULT_SYSTEM_PROMPT: &str = "You are a knowledge assistant for an organization. \
Answer questions using ONLY the provided context. Cite sources using [N] notation. \
If the context does not contain enough information, say so.";

/// Estimate token count using chars/4 heuristic.
fn estimate_tokens(text: &str) -> usize {
    text.len() / 4
}

/// Structured context augmenter with source citations and token budget.
pub struct StructuredAugmenter {
    system_prompt: String,
}

impl StructuredAugmenter {
    pub fn new(system_prompt: Option<String>) -> Self {
        Self {
            system_prompt: system_prompt.unwrap_or_else(|| DEFAULT_SYSTEM_PROMPT.into()),
        }
    }
}

impl Augmenter for StructuredAugmenter {
    fn name(&self) -> &str { "structured" }

    fn augment(
        &self,
        query: &str,
        results: &[SearchResult],
        budget: &TokenBudget,
    ) -> Result<AugmentedContext> {
        let start = std::time::Instant::now();

        let token_budget = budget.max_context_tokens.unwrap_or(usize::MAX);

        if results.is_empty() {
            return Ok(AugmentedContext {
                system_prompt: self.system_prompt.clone(),
                context: format!(
                    "No relevant context found.\n\nQuestion: {query}"
                ),
                sources: Vec::new(),
                metrics: AugmentationMetrics {
                    latency_ms: start.elapsed().as_millis() as u64,
                    token_estimate: 0,
                    token_budget,
                    sources_included: 0,
                    sources_truncated: 0,
                    augmenter_name: self.name().into(),
                },
            });
        }

        let mut context_parts: Vec<String> = Vec::new();
        let mut included: Vec<SearchResult> = Vec::new();
        let mut tokens_used: usize = 0;
        let mut truncated: usize = 0;

        // Header
        let header = "Context from organizational knowledge:\n";
        tokens_used += estimate_tokens(header);
        context_parts.push(header.into());

        // Pack sources by score until budget exhausted
        for (i, r) in results.iter().enumerate() {
            let source_info = match &r.entry.metadata.source_file {
                Some(f) => {
                    let chunk = r.entry.metadata.chunk_type.as_deref().unwrap_or("text");
                    format!("source: {f}, type: {chunk}")
                }
                None => "source: unknown".into(),
            };

            let entry_text = format!(
                "\n[{}] (score: {:.2}, {})\n{}\n",
                i + 1,
                r.score,
                source_info,
                r.entry.content,
            );

            let entry_tokens = estimate_tokens(&entry_text);
            if tokens_used + entry_tokens > token_budget {
                truncated += 1;
                continue;
            }

            tokens_used += entry_tokens;
            context_parts.push(entry_text);
            included.push(r.clone());
        }

        // Footer with query
        let footer = format!(
            "\nAnswer the following question using ONLY the context above. \
             Cite sources using [N] notation.\nQuestion: {query}"
        );
        tokens_used += estimate_tokens(&footer);
        context_parts.push(footer);

        let sources_included = included.len();
        let remaining = results.len() - sources_included;
        truncated += remaining.saturating_sub(truncated); // count any we skipped

        Ok(AugmentedContext {
            system_prompt: self.system_prompt.clone(),
            context: context_parts.join(""),
            sources: included,
            metrics: AugmentationMetrics {
                latency_ms: start.elapsed().as_millis() as u64,
                token_estimate: tokens_used,
                token_budget,
                sources_included,
                sources_truncated: truncated,
                augmenter_name: self.name().into(),
            },
        })
    }
}
```

```rust
        query: &str,
        scope_id: &str,
        opts: Option<RetrievalOpts>,
        generate: bool,
    ) -> Result<RagResponse> {
        let pipeline_start = Instant::now();
        let trace_id = Uuid::now_v7();

        let mode = if generate { "ask" } else { "context" };
        let _span = tracing::info_span!("corvia.rag.query",
            %trace_id, scope_id, mode,
        ).entered();
        info!(query, scope_id, mode, "rag_query_started");

        // Build opts from config defaults if not provided
        let opts = opts.unwrap_or_else(|| RetrievalOpts {
            limit: self.config.default_limit,
            expand_graph: self.config.graph_expand,
            graph_depth: self.config.graph_depth,
            ..Default::default()
        });

        // Stage 1: Retrieve
        let retrieval = self.retriever.retrieve(query, scope_id, &opts).await?;

        // Stage 2: Augment
        let budget = {
            let max_tokens = if self.config.max_context_tokens > 0 {
                Some(self.config.max_context_tokens)
            } else {
                // Model-aware auto-sizing (D64)
                self.generator.as_ref().map(|g| {
                    let window = g.context_window();
                    (window as f32 * (1.0 - self.config.reserve_for_answer)) as usize
                })
            };
            TokenBudget {
                max_context_tokens: max_tokens,
                reserve_for_answer: self.config.reserve_for_answer,
            }
        };

        let augment_start = Instant::now();
        let _as = tracing::info_span!("corvia.rag.augment").entered();
        let augmented = self.augmenter.augment(query, &retrieval.results, &budget)?;
        let augment_ms = augment_start.elapsed().as_millis() as u64;

        info!(
            sources_included = augmented.metrics.sources_included,
            token_estimate = augmented.metrics.token_estimate,
            "rag_augmented"
        );

        if augmented.metrics.sources_truncated > 0 {
            tracing::warn!(
                truncated = augmented.metrics.sources_truncated,
                "rag_budget_exceeded"
            );
        }

        // Stage 3: Generate (only in ask mode)
        let (answer, gen_metrics) = if generate {
            let gen = self.generator.as_ref().unwrap();
            let gen_start = Instant::now();
            let _gs = tracing::info_span!("corvia.rag.generate",
                model = gen.name(),
            ).entered();

            let result = gen.generate(
                &augmented.system_prompt,
                &augmented.context,
            ).await?;

            let gen_ms = gen_start.elapsed().as_millis() as u64;
            info!(
                model = result.model,
                output_tokens = result.output_tokens,
                "rag_generated"
            );

            let metrics = GenerationMetrics {
                latency_ms: gen_ms,
                model: result.model,
                input_tokens: result.input_tokens,
                output_tokens: result.output_tokens,
            };
            (Some(result.text), Some(metrics))
        } else {
            (None, None)
        };

        let total_ms = pipeline_start.elapsed().as_millis() as u64;
        info!(total_latency_ms = total_ms, "rag_query_completed");

        let trace = PipelineTrace {
            trace_id,
            retrieval: retrieval.metrics,
            augmentation: AugmentationMetrics {
                latency_ms: augment_ms,
                ..augmented.metrics.clone()
            },
            generation: gen_metrics,
            total_latency_ms: total_ms,
        };

        Ok(RagResponse {
            answer,
            context: augmented,
            trace,
        })
    }
}
```

```rust
/// Create a RAG pipeline with auto-selected retriever.
/// Uses GraphExpandRetriever when a GraphStore is available, VectorRetriever otherwise.
pub fn create_rag_pipeline(
    store: Arc<dyn traits::QueryableStore>,
    engine: Arc<dyn traits::InferenceEngine>,
    graph: Option<Arc<dyn traits::GraphStore>>,
    generator: Option<Arc<dyn traits::GenerationEngine>>,
    config: &CorviaConfig,
) -> rag_pipeline::RagPipeline {
    let retriever_: Arc<dyn retriever::Retriever> = match graph {
        Some(g) => Arc::new(retriever::GraphExpandRetriever::new(
            store.clone(), engine.clone(), g, config.rag.graph_alpha,
        )),
        None => Arc::new(retriever::VectorRetriever::new(store.clone(), engine.clone())),
    };
    let augmenter_: Arc<dyn augmenter::Augmenter> = Arc::new(
        augmenter::StructuredAugmenter::new(config.rag.system_prompt.clone()),
    );
    rag_pipeline::RagPipeline::new(retriever_, augmenter_, generator, config.rag.clone())
}
```

```rust
async fn search_memories(
    State(state): State<Arc<AppState>>,
    Json(req): Json<SearchRequest>,
) -> std::result::Result<Json<SearchResponse>, (StatusCode, String)> {
    let limit = req.limit.unwrap_or(10);

    // Route through RAG pipeline if available (fixes ContextBuilder bypass bug)
    if let Some(rag) = &state.rag {
        let opts = corvia_kernel::rag_types::RetrievalOpts {
            limit,
            expand_graph: false, // backward compat: pure vector for search endpoint
            ..Default::default()
        };
        let response = rag.context(&req.query, &req.scope_id, Some(opts)).await
            .map_err(|e| (StatusCode::INTERNAL_SERVER_ERROR, format!("Search failed: {e}")))?;

        let results: Vec<SearchResultDto> = response.context.sources.into_iter().map(Into::into).collect();
        let count = results.len();
        return Ok(Json(SearchResponse { results, count }));
    }

    // Fallback: raw store search (no RAG pipeline configured)
    let query_embedding = state.engine.embed(&req.query).await
        .map_err(|e| (StatusCode::INTERNAL_SERVER_ERROR, format!("Embedding failed: {e}")))?;

    let results = state.store.search(&query_embedding, &req.scope_id, limit).await
        .map_err(|e| (StatusCode::INTERNAL_SERVER_ERROR, format!("Search failed: {e}")))?;

    let count = results.len();
    let results: Vec<SearchResultDto> = results.into_iter().map(Into::into).collect();

    Ok(Json(SearchResponse { results, count }))
}
```
