```rust
//! Rust module path → file path resolution via Cargo.toml + `mod` declarations.
//!
//! Builds a map from Rust module paths (e.g., `crate::config`) to file paths
//! (e.g., `crates/corvia-common/src/config.rs`) by parsing workspace layout
//! and scanning source files for `mod foo;` declarations.

use std::collections::HashMap;
use std::path::{Path, PathBuf};

/// Maps Rust module paths to their source file paths.
///
/// Built by scanning workspace Cargo.toml files and `mod` declarations.
/// Used by `resolve_rust_module_path()` to create cross-file import edges.
#[derive(Debug, Clone)]
pub struct RustModuleMap {
    /// Maps `"crate_name::module::submodule"` → `"relative/path/to/file.rs"`
    /// Paths are relative to the workspace root.
    module_to_file: HashMap<String, String>,
    /// Maps crate package name → crate root module path prefix
    /// e.g., `"corvia-common"` → `"corvia_common"`
    crate_to_module: HashMap<String, String>,
}

impl RustModuleMap {
    /// Build a module map from a workspace or crate root directory.
    ///
    /// Scans for `Cargo.toml` files, finds crate roots, and walks `mod`
    /// declarations to build the full module→file mapping.
    pub fn build(workspace_root: &Path) -> Self {
        let mut map = RustModuleMap {
            module_to_file: HashMap::new(),
            crate_to_module: HashMap::new(),
        };

        let cargo_toml = workspace_root.join("Cargo.toml");
        if !cargo_toml.exists() {
            return map;
        }

        let Ok(content) = std::fs::read_to_string(&cargo_toml) else {
            return map;
        };
        let Ok(doc) = content.parse::<toml::Table>() else {
            return map;
        };

        // Collect crate directories: workspace members or single crate
        let crate_dirs = if let Some(workspace) = doc.get("workspace").and_then(|w| w.as_table()) {
            Self::resolve_workspace_members(workspace, workspace_root)
        } else {
            // Single crate at workspace root
            vec![workspace_root.to_path_buf()]
        };

        for crate_dir in &crate_dirs {
            map.index_crate(crate_dir, workspace_root);
        }

        map
    }

    /// Resolve a Rust module path to a file path.
    ///
    /// Handles:
    /// - `crate::config` → look up in this crate's module tree
    /// - `super::foo` → resolve relative to current file's parent module
    /// - `corvia_common::config` → look up cross-crate reference
    ///
    /// Returns `None` if the path can't be resolved (external crate, std, etc.).
    pub fn resolve(
        &self,
        current_file: &str,
        module_path: &str,
    ) -> Option<String> {
        if module_path.starts_with("crate::") {
            let relative = &module_path["crate::".len()..];
            // Find which crate this file belongs to
            let crate_prefix = self.crate_prefix_for_file(current_file)?;
            let lookup = format!("{}::{}", crate_prefix, relative);
            self.module_to_file.get(&lookup).cloned()
        } else if module_path.starts_with("super::") {
            let relative = &module_path["super::".len()..];
            self.resolve_super(current_file, relative)
        } else if module_path == "crate" || module_path == "super" {
            // Bare `crate` or `super` — resolve to parent
            if module_path == "crate" {
                let crate_prefix = self.crate_prefix_for_file(current_file)?;
                self.module_to_file.get(&crate_prefix).cloned()
            } else {
                self.resolve_super(current_file, "")
            }
        } else {
            // Could be a cross-crate workspace reference like `corvia_common::config`
            // Try looking it up directly
            self.module_to_file.get(module_path).cloned()
                .or_else(|| {
                    // Try first segment as crate name
                    let first_sep = module_path.find("::")?;
                    let crate_module = &module_path[..first_sep];
                    let rest = &module_path[first_sep + 2..];
                    let lookup = format!("{}::{}", crate_module, rest);
                    self.module_to_file.get(&lookup).cloned()
                })
        }
    }

    /// Resolve `super::` path relative to the current file.
    fn resolve_super(&self, current_file: &str, relative: &str) -> Option<String> {
        // Get the parent module by looking at the file path
        let path = Path::new(current_file);
        let parent_dir = if path.file_stem().map(|s| s == "mod" || s == "lib" || s == "main").unwrap_or(false) {
            // mod.rs/lib.rs/main.rs → go up two levels
            path.parent()?.parent()?
        } else {
            // foo.rs → parent is already the module directory
            path.parent()?
        };

        if relative.is_empty() {
            // Just `super` — return the parent module file
            let mod_rs = parent_dir.join("mod.rs");
            let lib_rs = parent_dir.join("lib.rs");
            if mod_rs.exists() {
                return Some(mod_rs.to_string_lossy().into_owned());
            }
            if lib_rs.exists() {
                return Some(lib_rs.to_string_lossy().into_owned());
            }
            return None;
        }

        // `super::foo` → look for foo.rs or foo/mod.rs in parent dir
        let parts: Vec<&str> = relative.split("::").collect();
        let first = parts[0];
        let file_path = parent_dir.join(format!("{}.rs", first));
        let mod_path = parent_dir.join(first).join("mod.rs");

        if file_path.exists() {
            Some(file_path.to_string_lossy().into_owned())
        } else if mod_path.exists() {
            Some(mod_path.to_string_lossy().into_owned())
        } else {
            None
        }
    }

    /// Find the crate module prefix for a given file path.
    fn crate_prefix_for_file(&self, file_path: &str) -> Option<String> {
        // Find the longest matching crate root path
        let mut best_match: Option<(&str, usize)> = None;
        for (module_prefix, file) in &self.module_to_file {
            // Only check crate-root entries (no `::` in the prefix after initial)
            if !module_prefix.contains("::") {
                let crate_dir = Path::new(file).parent().unwrap_or(Path::new(""));
                let crate_dir_str = crate_dir.to_string_lossy();
                if file_path.starts_with(crate_dir_str.as_ref()) {
                    let len = crate_dir_str.len();
                    if best_match.map_or(true, |(_, best_len)| len > best_len) {
                        best_match = Some((module_prefix.as_str(), len));
                    }
                }
            }
        }
        best_match.map(|(prefix, _)| prefix.to_string())
    }

    /// Parse workspace members from `[workspace]` table.
    fn resolve_workspace_members(workspace: &toml::Table, root: &Path) -> Vec<PathBuf> {
        let mut dirs = Vec::new();
        if let Some(members) = workspace.get("members").and_then(|m| m.as_array()) {
            for member in members {
                if let Some(pattern) = member.as_str() {
                    // Handle glob patterns like "crates/*"
                    if pattern.contains('*') {
                        let base = root.join(pattern.split('*').next().unwrap_or(""));
                        if let Ok(entries) = std::fs::read_dir(&base) {
                            for entry in entries.flatten() {
                                let path = entry.path();
                                if path.join("Cargo.toml").exists() {
                                    dirs.push(path);
                                }
                            }
                        }
                    } else {
                        let path = root.join(pattern);
                        if path.join("Cargo.toml").exists() {
                            dirs.push(path);
                        }
                    }
                }
            }
        }
        dirs
    }

    /// Index a single crate: find its root file and walk `mod` declarations.
    fn index_crate(&mut self, crate_dir: &Path, workspace_root: &Path) {
        let cargo_toml = crate_dir.join("Cargo.toml");
        let Ok(content) = std::fs::read_to_string(&cargo_toml) else {
            return;
        };
        let Ok(doc) = content.parse::<toml::Table>() else {
            return;
        };

        // Get package name
        let pkg_name = doc
            .get("package")
            .and_then(|p| p.as_table())
            .and_then(|p| p.get("name"))
            .and_then(|n| n.as_str())
            .unwrap_or("");

        if pkg_name.is_empty() {
            return;
        }

        // Convert package name to module name (hyphens → underscores)
        let module_name = pkg_name.replace('-', "_");

        // Find crate root file
        let lib_rs = crate_dir.join("src/lib.rs");
        let main_rs = crate_dir.join("src/main.rs");
        let root_file = if lib_rs.exists() {
            lib_rs
        } else if main_rs.exists() {
            main_rs
        } else {
            return;
        };

        let root_relative = Self::relative_path(&root_file, workspace_root);

        // Register crate root
        self.module_to_file
            .insert(module_name.clone(), root_relative.clone());
        self.crate_to_module
            .insert(pkg_name.to_string(), module_name.clone());

        // Walk mod declarations from root file
        self.walk_mod_declarations(
            &root_file,
            &module_name,
            workspace_root,
        );
    }

    /// Recursively walk `mod foo;` declarations to discover module files.
    fn walk_mod_declarations(
        &mut self,
        file: &Path,
        module_prefix: &str,
        workspace_root: &Path,
    ) {
        let Ok(content) = std::fs::read_to_string(file) else {
            return;
        };

        let parent_dir = file.parent().unwrap_or(Path::new(""));

        for line in content.lines() {
            let trimmed = line.trim();
            // Match `mod foo;` or `pub mod foo;` (NOT `mod foo { }` which is inline)
            if let Some(mod_name) = Self::parse_mod_declaration(trimmed) {
                // Look for foo.rs or foo/mod.rs
                let file_path = parent_dir.join(format!("{}.rs", mod_name));
                let mod_path = parent_dir.join(&mod_name).join("mod.rs");

                let resolved = if file_path.exists() {
                    Some(file_path)
                } else if mod_path.exists() {
                    Some(mod_path)
                } else {
                    None
                };

                if let Some(resolved_path) = resolved {
                    let relative = Self::relative_path(&resolved_path, workspace_root);
                    let full_module = format!("{}::{}", module_prefix, mod_name);
                    self.module_to_file.insert(full_module.clone(), relative);

                    // Recurse into the discovered module
                    self.walk_mod_declarations(&resolved_path, &full_module, workspace_root);
                }
            }
        }
    }

    /// Parse a `mod foo;` declaration, returning the module name.
    /// Returns `None` for inline modules (`mod foo { ... }`).
    fn parse_mod_declaration(line: &str) -> Option<&str> {
        let trimmed = line.trim();
        let rest = if let Some(rest) = trimmed.strip_prefix("pub mod ") {
            rest
        } else if let Some(rest) = trimmed.strip_prefix("pub(crate) mod ") {
            rest
        } else if let Some(rest) = trimmed.strip_prefix("pub(super) mod ") {
            rest
        } else if let Some(rest) = trimmed.strip_prefix("mod ") {
            rest
        } else {
            return None;
        };

        // Must end with `;` (not `{` which is inline)
        let name = rest.trim_end_matches(';').trim();
        if rest.ends_with(';') && !name.is_empty() && name.chars().all(|c| c.is_alphanumeric() || c == '_') {
            Some(name)
        } else {
            None
        }
    }

    /// Compute a relative path from workspace root.
    fn relative_path(path: &Path, workspace_root: &Path) -> String {
        path.strip_prefix(workspace_root)
            .unwrap_or(path)
            .to_string_lossy()
            .into_owned()
    }
}
```

```rust
impl MarkdownChunker {
    // ... existing methods ...

    /// Extract cross-references from markdown content using pulldown-cmark.
    ///
    /// Finds:
    /// - Code spans with CamelCase identifiers (likely types/traits)
    /// - Links to file paths with known extensions
    /// - File paths referenced in fenced code blocks
    fn extract_references(
        source: &str,
        chunks: &[RawChunk],
        meta: &SourceMetadata,
    ) -> Vec<ChunkRelation> {
        use pulldown_cmark::{Event, Parser, Tag, TagEnd};

        let mut relations = Vec::new();
        let parser = Parser::new(source);

        let mut current_line: u32 = 1;
        let mut in_code_block = false;

        for (event, range) in parser.into_offset_iter() {
            // Track line number from byte offset
            current_line = source[..range.start].matches('\n').count() as u32 + 1;

            match event {
                Event::Code(code) => {
                    // Inline code span: `QueryableStore`, `LiteStore`, etc.
                    let text = code.as_ref().trim();
                    if Self::is_meaningful_symbol(text) {
                        if let Some(chunk) = Self::find_chunk_for_line(chunks, current_line) {
                            relations.push(ChunkRelation {
                                from_source_file: meta.file_path.clone(),
                                from_start_line: chunk.start_line,
                                relation: "references".to_string(),
                                to_file: String::new(),
                                to_name: Some(text.to_string()),
                            });
                        }
                    }
                }
                Event::Start(Tag::Link { dest_url, .. }) => {
                    let url = dest_url.as_ref();
                    if Self::is_file_path_reference(url) {
                        // Normalize the path (strip ../ prefixes, anchors, etc.)
                        let clean_path = url.split('#').next().unwrap_or(url);
                        let clean_path = clean_path.trim_start_matches("./");
                        if !clean_path.is_empty() {
                            if let Some(chunk) = Self::find_chunk_for_line(chunks, current_line) {
                                relations.push(ChunkRelation {
                                    from_source_file: meta.file_path.clone(),
                                    from_start_line: chunk.start_line,
                                    relation: "references".to_string(),
                                    to_file: clean_path.to_string(),
                                    to_name: None,
                                });
                            }
                        }
                    }
                }
                Event::Start(Tag::CodeBlock(_)) => {
                    in_code_block = true;
                }
                Event::End(TagEnd::CodeBlock) => {
                    in_code_block = false;
                }
                Event::Text(text) if in_code_block => {
                    // Scan code block content for file path references
                    for word in text.as_ref().split_whitespace() {
                        if Self::is_file_path_reference(word) {
                            let clean = word.trim_matches(|c: char| c == '"' || c == '\'' || c == '`');
                            if let Some(chunk) = Self::find_chunk_for_line(chunks, current_line) {
                                relations.push(ChunkRelation {
                                    from_source_file: meta.file_path.clone(),
                                    from_start_line: chunk.start_line,
                                    relation: "references".to_string(),
                                    to_file: clean.to_string(),
                                    to_name: None,
                                });
                            }
                        }
                    }
                }
                _ => {}
            }
        }

        // Deduplicate relations (same from_start_line + to_file + to_name)
        relations.sort_by(|a, b| {
            a.from_start_line.cmp(&b.from_start_line)
                .then(a.to_file.cmp(&b.to_file))
                .then(a.to_name.cmp(&b.to_name))
        });
        relations.dedup_by(|a, b| {
            a.from_start_line == b.from_start_line
                && a.to_file == b.to_file
                && a.to_name == b.to_name
        });

        relations
    }

    /// Check if a code span contains a meaningful symbol reference.
    /// Must be CamelCase (type/trait name) and not a Rust primitive.
    fn is_meaningful_symbol(text: &str) -> bool {
        if text.len() < 3 {
            return false;
        }

        // Must start with uppercase (CamelCase)
        if !text.starts_with(|c: char| c.is_ascii_uppercase()) {
            return false;
        }

        // Must be a valid identifier
        if !text.chars().all(|c| c.is_alphanumeric() || c == '_') {
            return false;
        }

        // Skip common Rust primitives and std types
        const SKIP_TYPES: &[&str] = &[
            "String", "Vec", "HashMap", "HashSet", "BTreeMap", "BTreeSet",
            "Option", "Result", "Box", "Arc", "Mutex", "RwLock", "Rc",
            "Cell", "RefCell", "Pin", "Future", "Stream", "Iterator",
            "Display", "Debug", "Clone", "Copy", "Send", "Sync", "Default",
            "From", "Into", "TryFrom", "TryInto", "AsRef", "AsMut",
            "Deref", "DerefMut", "Drop", "Fn", "FnMut", "FnOnce",
            "Error", "Ord", "PartialOrd", "Eq", "PartialEq", "Hash",
            "Sized", "Unpin", "Serialize", "Deserialize",
            "None", "Some", "Ok", "Err", "Self", "TRUE", "FALSE",
        ];
        !SKIP_TYPES.contains(&text)
    }

    /// Check if a string looks like a file path reference.
    fn is_file_path_reference(s: &str) -> bool {
        // Must not be an HTTP URL
        if s.starts_with("http://") || s.starts_with("https://") {
            return false;
        }
        // Must contain a known source file extension
        const KNOWN_EXTENSIONS: &[&str] = &[
            ".rs", ".ts", ".tsx", ".js", ".jsx", ".py",
            ".toml", ".yaml", ".yml", ".json", ".md",
        ];
        KNOWN_EXTENSIONS.iter().any(|ext| s.ends_with(ext))
    }

    /// Find which chunk a given line number falls in.
    fn find_chunk_for_line(chunks: &[RawChunk], line: u32) -> Option<&RawChunk> {
        chunks.iter().find(|c| line >= c.start_line && line <= c.end_line)
            .or_else(|| chunks.last()) // fallback to last chunk if line is after all chunks
    }
}
```

```rust
    /// Discover cross-file relations by tracking type/trait definitions and usages.
    ///
    /// Pass 1: Collect definitions (structs, traits, enums) from chunk metadata.
    /// Pass 2: Find cross-file usages of those definitions in chunk content.
    /// Returns additional `ChunkRelation` entries with relation type `"uses"`.
    pub fn discover_cross_file_relations(
        chunks: &[ProcessedChunk],
    ) -> Vec<ChunkRelation> {
        use std::collections::{HashMap, HashSet};

        // Standard library types to skip
        const SKIP_TYPES: &[&str] = &[
            "String", "Vec", "HashMap", "HashSet", "BTreeMap", "BTreeSet",
            "Option", "Result", "Box", "Arc", "Mutex", "RwLock", "Rc",
            "Cell", "RefCell", "Pin", "Future", "Stream", "Iterator",
            "Display", "Debug", "Clone", "Copy", "Send", "Sync", "Default",
            "From", "Into", "TryFrom", "TryInto", "AsRef", "AsMut",
            "Deref", "DerefMut", "Drop", "Fn", "FnMut", "FnOnce",
            "Error", "Ord", "PartialOrd", "Eq", "PartialEq", "Hash",
            "Sized", "Unpin", "Serialize", "Deserialize",
            "None", "Some", "Ok", "Err", "Self", "TRUE", "FALSE",
            "Path", "PathBuf", "File", "Read", "Write", "Seek",
            "Command", "Output", "ExitStatus",
        ];
        let skip_set: HashSet<&str> = SKIP_TYPES.iter().copied().collect();

        // Pass 1: Collect definitions
        // Maps symbol_name → Vec<(source_file, start_line)>
        let mut definitions: HashMap<String, Vec<(String, u32)>> = HashMap::new();

        for chunk in chunks {
            let chunk_type = &chunk.chunk_type;
            // Only look at definition-like chunk types
            if !matches!(
                chunk_type.as_str(),
                "struct_item" | "trait_item" | "enum_item" | "impl_item" | "mod_item"
                    | "struct" | "trait" | "enum" | "impl" | "mod"
            ) {
                continue;
            }

            // Extract the name from the first line of content
            if let Some(name) = Self::extract_definition_name(&chunk.content) {
                if name.len() >= 3
                    && name.starts_with(|c: char| c.is_ascii_uppercase())
                    && !skip_set.contains(name.as_str())
                {
                    definitions
                        .entry(name)
                        .or_default()
                        .push((chunk.metadata.source_file.clone(), chunk.start_line));
                }
            }
        }

        // Remove symbols defined in too many files (too generic)
        let definitions: HashMap<String, Vec<(String, u32)>> = definitions
            .into_iter()
            .filter(|(_, locs)| {
                let unique_files: HashSet<_> = locs.iter().map(|(f, _)| f.as_str()).collect();
                unique_files.len() < 10
            })
            .collect();

        // Pass 2: Find cross-file usages
        let mut relations = Vec::new();

        for chunk in chunks {
            let chunk_file = &chunk.metadata.source_file;

            for (symbol, def_locations) in &definitions {
                // Skip if this chunk defines the symbol (same file + same start line)
                let is_self_def = def_locations
                    .iter()
                    .any(|(f, l)| f == chunk_file && *l == chunk.start_line);
                if is_self_def {
                    continue;
                }

                // Check if chunk content contains the symbol as a word boundary match
                if Self::content_references_symbol(&chunk.content, symbol) {
                    // Find cross-file definitions only
                    for (def_file, def_line) in def_locations {
                        if def_file != chunk_file {
                            relations.push(ChunkRelation {
                                from_source_file: chunk_file.clone(),
                                from_start_line: chunk.start_line,
                                relation: "uses".to_string(),
                                to_file: def_file.clone(),
                                to_name: Some(symbol.clone()),
                            });
                        }
                    }
                }
            }
        }

        // Deduplicate
        relations.sort_by(|a, b| {
            a.from_source_file
                .cmp(&b.from_source_file)
                .then(a.from_start_line.cmp(&b.from_start_line))
                .then(a.to_file.cmp(&b.to_file))
                .then(a.to_name.cmp(&b.to_name))
        });
        relations.dedup_by(|a, b| {
            a.from_source_file == b.from_source_file
                && a.from_start_line == b.from_start_line
                && a.to_file == b.to_file
                && a.to_name == b.to_name
        });

        relations
    }

    /// Extract the defined symbol name from a chunk's content.
    ///
    /// Looks for patterns like:
    /// - `pub struct FooBar` → "FooBar"
    /// - `pub trait MyTrait` → "MyTrait"
    /// - `pub enum MyEnum` → "MyEnum"
    fn extract_definition_name(content: &str) -> Option<String> {
        let first_line = content.lines().next()?;
        let trimmed = first_line.trim();

        // Try to match: [pub [(...)] ] (struct|trait|enum) Name
        let keywords = ["struct ", "trait ", "enum "];
        for keyword in &keywords {
            if let Some(pos) = trimmed.find(keyword) {
                let after = &trimmed[pos + keyword.len()..];
                let name: String = after
                    .chars()
                    .take_while(|c| c.is_alphanumeric() || *c == '_')
                    .collect();
                if !name.is_empty() {
                    return Some(name);
                }
            }
        }

        None
    }

    /// Check if content references a symbol at a word boundary.
    /// Avoids matching substrings (e.g., "Store" inside "LiteStore").
    fn content_references_symbol(content: &str, symbol: &str) -> bool {
        let mut start = 0;
        while let Some(pos) = content[start..].find(symbol) {
            let abs_pos = start + pos;
            let before_ok = abs_pos == 0
                || !content.as_bytes()[abs_pos - 1].is_ascii_alphanumeric()
                    && content.as_bytes()[abs_pos - 1] != b'_';
            let after_pos = abs_pos + symbol.len();
            let after_ok = after_pos >= content.len()
                || !content.as_bytes()[after_pos].is_ascii_alphanumeric()
                    && content.as_bytes()[after_pos] != b'_';
            if before_ok && after_ok {
                return true;
            }
            start = abs_pos + 1;
        }
        false
    }
```
