# M4.5: CUDA Toggle Implementation Plan

> **Status:** Shipped (v0.4.0)

**Goal:** Add `corvia-workspace enable/disable cuda` to toggleably accelerate ONNX inference with NVIDIA GPUs.

**Architecture:** Extend the existing `corvia-workspace.sh` toggle script with `enable_cuda`/`disable_cuda` functions. GPU detection via `nvidia-smi`. CUDA binary downloaded from GitHub releases (fallback: source build with `--features cuda`). No Dockerfile or devcontainer.json changes.

**Step 1: Add the CUDA helper functions**

Insert the following block after the `disable_surrealdb()` function (after line 137, before `# --- Status ---`):

```bash
# --- CUDA ---

# Detect GPU info string for status display
gpu_info() {
    if command -v nvidia-smi &>/dev/null; then
        nvidia-smi --query-gpu=name,driver_version,memory.total \
            --format=csv,noheader 2>/dev/null | head -1 | sed 's/, / /g'
    else
        echo "not detected"
    fi
}

install_cuda_toolkit() {
    echo "  Installing CUDA toolkit (this may take a few minutes)..."

    # Add NVIDIA apt repository for Debian
    # Works for Trixie (base image) and WSL2 Ubuntu
    local arch
    arch=$(dpkg --print-architecture)

    # Install prerequisites
    apt-get update -qq
    apt-get install -y -qq gnupg2 software-properties-common >/dev/null 2>&1

    # Try distro-specific keyring first, fall back to generic ubuntu2404
    local distro_id distro_version_id keyring_url repo_url
    distro_id=$(. /etc/os-release && echo "$ID")
    distro_version_id=$(. /etc/os-release && echo "$VERSION_ID")

    if [ "$distro_id" = "debian" ]; then
        # Debian Trixie — use debian12 (closest supported)
        keyring_url="https://developer.download.nvidia.com/compute/cuda/repos/debian12/${arch}/cuda-keyring_1.1-1_all.deb"
        repo_url="deb [signed-by=/usr/share/keyrings/cuda-archive-keyring.gpg] https://developer.download.nvidia.com/compute/cuda/repos/debian12/${arch}/ /"
    else
        # Ubuntu / WSL2 Ubuntu
        local ver="${distro_version_id//./}"
        keyring_url="https://developer.download.nvidia.com/compute/cuda/repos/ubuntu${ver}/${arch}/cuda-keyring_1.1-1_all.deb"
        repo_url="deb [signed-by=/usr/share/keyrings/cuda-archive-keyring.gpg] https://developer.download.nvidia.com/compute/cuda/repos/ubuntu${ver}/${arch}/ /"
    fi

    curl -fsSL -o /tmp/cuda-keyring.deb "$keyring_url"
    dpkg -i /tmp/cuda-keyring.deb
    rm -f /tmp/cuda-keyring.deb
    apt-get update -qq

    # Install CUDA toolkit and cuDNN
    apt-get install -y -qq cuda-toolkit libcudnn9-dev-cuda-12 >/dev/null 2>&1 || \
    apt-get install -y -qq cuda-toolkit-12-6 libcudnn9-dev-cuda-12 >/dev/null 2>&1 || {
        echo "  Error: Failed to install CUDA toolkit. Check network and apt sources."
        exit 1
    }

    echo "  CUDA toolkit installed."
}

restart_inference() {
    if pgrep -f "corvia-inference" >/dev/null 2>&1; then
        echo "  Restarting corvia-inference..."
        pkill -f "corvia-inference" 2>/dev/null || true
        sleep 1
        # The server will be restarted by corvia serve on next request,
        # or user can run corvia-workspace status to verify
    fi
}

enable_cuda() {
    echo "Enabling CUDA..."

    # 1. Check for NVIDIA GPU
    if ! command -v nvidia-smi &>/dev/null; then
        # Check WSL2 userspace driver
        if [ -f /usr/lib/wsl/lib/libcuda.so ]; then
            echo "  WSL2 CUDA userspace detected, but nvidia-smi not in PATH."
            echo "  Add /usr/lib/wsl/lib to PATH or install nvidia-utils package."
        else
            echo "  Error: No NVIDIA GPU detected."
            echo "  Ensure NVIDIA drivers are installed on the host and GPU passthrough is configured."
            echo "  For WSL2: install NVIDIA GPU driver on Windows, then restart WSL."
        fi
        exit 1
    fi

    echo "  GPU detected: $(gpu_info)"

    # 2. Install CUDA toolkit if nvcc is missing
    if ! command -v nvcc &>/dev/null; then
        install_cuda_toolkit
    else
        echo "  CUDA toolkit already installed: $(nvcc --version | grep release | awk '{print $6}')"
    fi

    # 3. Download CUDA-enabled binary
    local release_url="https://github.com/chunzhe10/corvia/releases/latest/download"
    local cuda_binary="corvia-inference-cuda-linux-amd64"

    echo "  Downloading CUDA-enabled corvia-inference binary..."
    if curl -fsSL -o /usr/local/bin/corvia-inference-cuda "$release_url/$cuda_binary"; then
        chmod +x /usr/local/bin/corvia-inference-cuda
    else
        echo "  Download failed — building from source with CUDA feature..."
        local corvia_src="$WORKSPACE_ROOT/repos/corvia"
        if [ ! -d "$corvia_src" ]; then
            echo "  Error: corvia source not found at $corvia_src"
            exit 1
        fi
        cd "$corvia_src"
        cargo install --path crates/corvia-inference --features cuda
        cp "$(which corvia-inference)" /usr/local/bin/corvia-inference-cuda
        cd "$WORKSPACE_ROOT"
    fi

    # 4. Backup CPU binary and swap in CUDA binary
    if [ ! -f /usr/local/bin/corvia-inference-cpu ]; then
        cp /usr/local/bin/corvia-inference /usr/local/bin/corvia-inference-cpu
    fi
    cp /usr/local/bin/corvia-inference-cuda /usr/local/bin/corvia-inference

    # 5. Set flag and restart
    set_flag cuda enabled
    restart_inference

    echo "  CUDA enabled. corvia-inference now uses GPU-accelerated ONNX inference."
}

disable_cuda() {
    echo "Disabling CUDA..."

    # 1. Restore CPU binary
    if [ -f /usr/local/bin/corvia-inference-cpu ]; then
        cp /usr/local/bin/corvia-inference-cpu /usr/local/bin/corvia-inference
    else
        echo "  Warning: CPU binary backup not found. corvia-inference unchanged."
    fi

    # 2. Set flag and restart
    set_flag cuda disabled
    restart_inference

    echo "  CUDA disabled. corvia-inference reverted to CPU-only ONNX inference."
}
```
